async function loadProducts(){
  const {data}=await sb.from('products').select('*').order('name');

    list.innerHTML=data.filter(p=>p.name.toLowerCase().includes(filter.toLowerCase()))

function draw(){
  ctx.clearRect(0,0,cv.width,cv.height);
  if(img.src) ctx.drawImage(img,0,0,cv.width,cv.height);
  const txt=document.getElementById('txt').value||'';
  const font=document.getElementById('font').value;
  const size=+document.getElementById('size').value;
  const dir=document.getElementById('dir').value;
  const valign=document.getElementById('valign').value;
  const voff=+document.getElementById('voff').value;
  ctx.fillStyle='rgb(128,128,128)'; ctx.textAlign='center'; ctx.font=`${size}px ${font}`;
  let x=cv.width/2, y=cv.height/2+voff;
  if(valign==='top') y=size+10+voff;
  if(valign==='bottom') y=cv.height-10+voff;
  if(dir==='horizontal'){ctx.fillText(txt,x,y);}
  else{
    const chars=txt.split(''); const step=size*1.1;
    let startY=y-(chars.length-1)*step/2;
    if(dir==='vertical-up'){chars.reverse();}
    chars.forEach((c,i)=>{ctx.fillText(c,x,startY+i*step);});
  }
}

async function saveOrder(status){
  const pngBlob=await new Promise(r=>cv.toBlob(r));
  const pngName=`${crypto.randomUUID()}.png`;
  await sb.storage.from('previews').upload(pngName,pngBlob,{upsert:true});
  const pngUrl=`${SUPABASE_URL}/storage/v1/object/public/previews/${pngName}`;
  let pdfUrl=null;
  if(status==='submitted'){
    const pdfDoc=await PDFLib.PDFDocument.create();
    const page=pdfDoc.addPage([cv.width,cv.height]);
    const pngBytes=await pngBlob.arrayBuffer();
    const pngImg=await pdfDoc.embedPng(pngBytes);
    page.drawImage(pngImg,{x:0,y:0,width:cv.width,height:cv.height});
    const pdfBytes=await pdfDoc.save();
    const pdfName=`${crypto.randomUUID()}.pdf`;
    await sb.storage.from('previews').upload(pdfName,new Blob([pdfBytes]),{upsert:true});
    pdfUrl=`${SUPABASE_URL}/storage/v1/object/public/previews/${pdfName}`;
  }
  await sb.from('orders').insert({
    user_id:user.id,
    customer_name:document.getElementById('custName').value,
    customer_phone:document.getElementById('custPhone').value,
    due_date:document.getElementById('dueDate').value,
    product_id:product?.id,
    preview_text:document.getElementById('txt').value,
    font_family:document.getElementById('font').value,
    font_size:+document.getElementById('size').value,
    text_direction:document.getElementById('dir').value,
    vertical_align:document.getElementById('valign').value,
    vertical_offset:+document.getElementById('voff').value,
    preview_png_url:pngUrl,
    preview_pdf_url:pdfUrl,
    status
  });
  alert(status==='draft'?'Saved!':'Submitted!');
}
